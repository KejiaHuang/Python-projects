import statsmodels.api as sm
import numpy as np
import pandas as pd
from scipy import linalg
from scipy.optimize import minimize

''' functions for the regression and optimization'''
def dot(x,y,z):
    d=x.dot(y)
    return d.dot(z)
    
def reg(x,y): #regression
    x=sm.add_constant(x)
    res=sm.OLS(y,x).fit()
    return res.params.values
    
def opt1(sig,l,r,w0): #optimization(benchmark,minVar,expected r=15%)
    def func1(w):
        ww=np.array([w])
        return dot(ww,sig,ww.T)+l*dot((ww.T-w0).T,Q,(ww.T-w0))
    cons=({'type':'eq','fun':lambda w: np.array([np.array(w).dot(r)[0]-15/252])},
         {'type':'eq','fun':lambda w: np.array([sum(w)-1])})
    res1=minimize(func1,13*[1/13],constraints=cons,bounds=[[-2,2]]*13)
    return res1
    
def opt2(sig,l,r,b,w0): #optimization(max return, with target beta)
    def func2(w):
        ww=np.array([w])
        return -ww.dot(r)/100+l*dot((ww.T-w0).T,Q,(ww.T-w0))
    cons=({'type':'eq','fun':lambda w: np.array([np.array(w).dot(bm)[0]-b])},
         {'type':'eq','fun':lambda w: np.array([sum(w)-1])})
    res2=minimize(func2,13*[1/13],constraints=cons,bounds=[[-2,2]]*13)
    return res2

''' Main function, return the daily return of all the strategies'''
#t1:time interval of rebalancing,t2:time interval of historical data used
#l:lambda, to limit the turnover,bT:target beta
def project_function(t1,t2,l): 
    r1=np.zeros((len(R)-t2,1)) #actual retrun of the strategy(min var)
    r2_1=np.zeros((len(R)-t2,1))#actual retrun of the strategy(max r)
    r2_2=np.zeros((len(R)-t2,1))
    r2_3=np.zeros((len(R)-t2,1))
    r0=R['SPY'][t2:len(R)]*100#return of market(SPY)
    for i in range (t2,len(R)):
        if (i-t2)%t1==0:
            x=X[i-t2:i] #x: 3 factors
            for j in range (0,13):
                y=R.iloc[:,j]-ff['RF']/252
                y=y[i-t2:i]*100 #y=r-rf
                res=reg(x,y)
                r=sm.add_constant(X[i:i+1]).dot(res) #expected return
                er[j]=r.values+ff['RF'][i-1:i]*100/252
                # c is used to get the beta with market(spy)
                c=np.zeros((2,t2))
                c[0]=R['SPY'][i-t2:i].T*100
                c[1]=(R.iloc[i-t2:i,j]).T*100
                bm[j]=np.cov(c)[1,0]/np.var(c[0])
            sig=np.cov((R[i-t2:i]*100).T)
            o1=opt1(sig,l,er,w1)
            o2_1=opt2(sig,l,er,0.5,w2_1)
            o2_2=opt2(sig,l,er,1,w2_2)
            o2_3=opt2(sig,l,er,1.5,w2_3)
            for k in range(0,13):
                w1[k]=o1.x[k]
                w2_1[k]=o2_1.x[k]
                w2_2[k]=o2_2.x[k]
                w2_3[k]=o2_3.x[k]
        r1[i-t2]=R[i:i+1].dot(w1)*100
        r2_1[i-t2]=R[i:i+1].dot(w2_1)*100
        r2_2[i-t2]=R[i:i+1].dot(w2_2)*100
        r2_3[i-t2]=R[i:i+1].dot(w2_3)*100
        
    F1=pd.DataFrame(r1)
    F2_1=pd.DataFrame(r2_1)
    F2_2=pd.DataFrame(r2_2)
    F2_3=pd.DataFrame(r2_3)
    F1=F1.set_index(r0.index)
    F2_1=F2_1.set_index(r0.index)
    F2_2=F2_2.set_index(r0.index)
    F2_3=F2_3.set_index(r0.index)
    Return=pd.DataFrame({'Spy500':r0,
                    'minVar':F1[0],
                    'maxR,bm=0.5':F2_1[0],
                    'maxR,bm=1':F2_2[0],
                    'maxR,bm=1.5':F2_3[0]})
    return Return

import matplotlib.pyplot as plt
from scipy import stats
import numpy as np
import pandas as pd
from math import sqrt, log, exp

#Cumulated return
def cumulated_R(Daily_R):
    A=pd.DataFrame(np.zeros((len(Daily_R),5)))
    for i in range (0,len(A)):
        A.iloc[i,:]=np.array(np.sum(Daily_R[0:i+1]))
    A=A.set_index(Daily_R.index)
    return A

#max 10 day drawdown
def max_10_dd(r):
    dd=pd.DataFrame(np.zeros((len(r)-9,5)))
    for i in range (9,len(r)):
        dd.iloc[i-9,:]=np.array(r.iloc[i,:]-r.iloc[i-9,:])
    return np.array(np.max(-dd))
    
#volatility (standard deviation)
def vol(r):
    v=np.std(r/100)
    return v
    
#skewness
def skew(r):
    skew=stats.skew(r/100)
    return skew

#kurtosis
def kurt(r):
    kurt=stats.kurtosis(r/100)
    return kurt

#sharp ratio
def sharp(r):
    s1=np.mean(r['Spy500']/100-ff['RF'][len(ff)-len(r):len(ff)]/252)/vol(r['Spy500'])
    s2=np.mean(r['maxR,bm=0.5']/100-ff['RF'][len(ff)-len(r):len(ff)]/252)/vol(r['maxR,bm=0.5'])
    s3=np.mean(r['maxR,bm=1']/100-ff['RF'][len(ff)-len(r):len(ff)]/252)/vol(r['maxR,bm=1'])
    s4=np.mean(r['maxR,bm=1.5']/100-ff['RF'][len(ff)-len(r):len(ff)]/252)/vol(r['maxR,bm=1.5'])
    s5=np.mean(r['minVar']/100-ff['RF'][len(ff)-len(r):len(ff)]/252)/vol(r['minVar'])
    return np.array([s1,s2,s3,s4,s5])
#Var
def var(a,r):
    x=int((1-a)*len(r))
    rr1=r['Spy500'].sort_values()
    rr2_1=r['maxR,bm=0.5'].sort_values()
    rr2_2=r['maxR,bm=1'].sort_values()
    rr2_3=r['maxR,bm=1.5'].sort_values()
    rr3=r['minVar'].sort_values()
    return np.array([rr1[x],rr2_1[x],rr2_2[x],rr2_3[x],rr3[x]])
    
#Cvar
def cvar(a,r):
    x=int((1-a)*len(r))
    rr1=r['Spy500'].sort_values()
    rr2_1=r['maxR,bm=0.5'].sort_values()
    rr2_2=r['maxR,bm=1'].sort_values()
    rr2_3=r['maxR,bm=1.5'].sort_values()
    rr3=r['minVar'].sort_values()
    return np.array([np.mean(rr1[0:x]),np.mean(rr2_1[0:x]),np.mean(rr2_2[0:x]),np.mean(rr2_3[0:x]),np.mean(rr3[0:x])])

def dist(r):
    for i in range(0,5):
        rr=r.iloc[:,i]
        f=np.histogram(rr,bins=100)
        p=f[0]/np.sum(f[0])
        x1=f[1][0:100]
        x2=f[1][1:101]
        xx=(x1+x2)/2
        plt.plot(xx,p,label=r.columns[i])
    return
